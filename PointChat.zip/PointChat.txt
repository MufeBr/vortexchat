        const loginBtn = document.getElementById("loginBtn");
        const usernameInput = document.getElementById("username");
        const errorMessage = document.getElementById("error-message");

        const chatContainer = document.getElementById("chat-container");
        const chatBox = document.getElementById("chatBox");
        const chatInput = document.getElementById("chatInput");
        const chatSend = document.getElementById("chatSend");
        const skipBtn = document.getElementById("skipBtn");

        const emojiBtn = document.getElementById("emojiBtn");
        const emojiPicker = document.getElementById("emojiPicker");

        /* LOGIN */
        loginBtn.onclick = () => {
            username = usernameInput.value.trim();
            if (!username) {
                errorMessage.style.display = "block";
                return;
            }

            loginScreen.style.display = "none";
            chatContainer.style.display = "flex";
            document.querySelector(".desktop-header").style.display = "flex";
            document.querySelector(".mobile-header").style.display = "flex";

            listenMessages();
        };

        /* ADICIONAR MENSAGEM */
        function addMsg(type, txt) {
            const div = document.createElement("div");
            div.className = type;
            div.innerHTML = txt;
            chatBox.appendChild(div);
            chatBox.scrollTop = chatBox.scrollHeight;
        }

        /* ENVIAR MENSAGEM */
        chatSend.onclick = () => {
            const txt = chatInput.value.trim();
            if (!txt) return;

            db.ref("messages").push({
                user: username,
                text: txt,
                time: Date.now()
            });

            chatInput.value = "";
        };

        chatInput.addEventListener("keydown", e => {
            if (e.key === "Enter") chatSend.click();
        });

        /* RECEBER MENSAGENS */
        function listenMessages() {
            db.ref("messages").on("child_added", snap => {
                const msg = snap.val();

                if (msg.user === username) {
                    addMsg("msg-me", "<b>" + msg.user + ":</b> " + msg.text);
                } else {
                    addMsg("msg-them", "<b>" + msg.user + ":</b> " + msg.text);
                }
            });
        }

        /* SKIP */
        skipBtn.onclick = () => {
            chatBox.innerHTML = "";
            addMsg("system", "Voc√™ pulou a conversa!");
        };

        /* EMOJIS */
        emojiBtn.onclick = () => {
            emojiPicker.style.display = emojiPicker.style.display === "none" ? "flex" : "none";
        };

        document.querySelectorAll(".emoji").forEach(em => {
            em.onclick = () => {
                chatInput.value += em.textContent;
                chatInput.focus();
            };
        });
